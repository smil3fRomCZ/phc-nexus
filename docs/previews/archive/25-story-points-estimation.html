Schema::table('tasks', function (Blueprint $table) {
    $table->unsignedSmallInteger('story_points')->nullable();
    $table->decimal('estimated_hours', 6, 2)->nullable();
});
// Task model accessor
// SP x 4h pokud SP nastaveno, jinak manualni estimated_hours
public function getComputedEstimatedHoursAttribute(): ?float
{
    if ($this->story_points !== null) {
        return $this->story_points * 4;
    }
    return $this->estimated_hours;
}
